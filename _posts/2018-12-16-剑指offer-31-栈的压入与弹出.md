## 思路
    
 借助一个辅助栈s，将pushS栈中的数字从栈底往栈顶往s中push。
在push的过程中如果该数等于popS的栈底元素则s pop。最终如果s
为空则为true。


## 代码

	class Solution
	 {
	  public:
        bool IsPopOrder(vector<int> pushV,vector<int> popV)
        {
      	 	 if(pushV.size() == 0)
      	 	 {
      	 	     return false;
      	 	 }
      	 	 vector<int> s;
      	 	 for(int i = 0 , j = 0; i < pushV.size() ;  )
      	        {
      		      s.push_back(pushV[i++]);
      		      while(popV[j] == s.back() && j < popV.size())
      		      {
      		          s.pop_back();
      		          j++;
      		      }
      		   }
      	  return s.empty();
    	}
	};
